        };
        
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        // Inicializa o Mapa focado em Alphaville/SP (ajuste se necessário)
        const map = L.map('map-global').setView([-23.49, -46.85], 13);
        L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png').addTo(map);

        let activeLayers = [];

        // 1. Buscar Resumo da Frota (Odômetros)
        db.ref('vehicles').on('value', snapshot => {
            const data = snapshot.val();
            const container = document.getElementById('fleet-stats');
            container.innerHTML = '';
            for(let id in data) {
                container.innerHTML += `
                    <div class="car-badge">
                        <small>${id}</small>
                        <span>${data[id].odometer.toFixed(1)} km</span>
                    </div>`;
            }
        });

        // 2. Buscar Viagens e Desenhar no Mapa
        db.ref('trips').orderByChild('date').limitToLast(20).on('value', snapshot => {
            const data = snapshot.val();
            const container = document.getElementById('trips-list');
            container.innerHTML = '';
            
            // Inverter para mostrar as mais recentes primeiro
            const trips = Object.keys(data).reverse();

            trips.forEach(key => {
                const t = data[key];
                const card = document.createElement('div');
                card.className = 'trip-card';
                card.innerHTML = `
                    <b>${t.car} - ${t.driver}</b>
                    <small>📅 ${t.date}</small><br>
                    <small>📏 ${t.km} km | ⏱️ ${t.time}</small>
                `;
                
                card.onclick = () => showTripOnMap(t.path);
                container.appendChild(card);
            });
        });

        function showTripOnMap(path) {
            // Limpa linhas anteriores
            activeLayers.forEach(layer => map.removeLayer(layer));
            activeLayers = [];

            if (!path || path.length < 2) return alert("Esta viagem não registrou trajeto.");

            // Desenha a Linha (Path) que veio do Banco de Dados
            const line = L.polyline(path, { color: '#f39c12', weight: 6, opacity: 0.8 }).addTo(map);
            const startMarker = L.circleMarker(path[0], { radius: 6, color: '#27ae60', fillOpacity: 1 }).addTo(map);
            const endMarker = L.circleMarker(path[path.length - 1], { radius: 6, color: '#e74c3c', fillOpacity: 1 }).addTo(map);

            activeLayers.push(line, startMarker, endMarker);
            map.fitBounds(line.getBounds(), { padding: [50, 50] });
        }
